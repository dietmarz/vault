----
spring:
    datasource.url: "jdbc:h2:mem:test;DB_CLOSE_DELAY=-1" # keep DB open
    jpa:

        properties.hibernate.show_sql: true

  org.hibernate.sql: DEBUG
  org.hibernate.type: DEBUG
