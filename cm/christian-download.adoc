== Convert
----
# Rename downloaded files
   removeSpacesInFiles
   ls > move.sh
   # Bearbeiten der namen in Intellij mit <shift><alt><insert> im Insertmode oder mit gedrückter Mittelmaustaste

# Format: yyyy-mm-dd-N-18mt3
# Format All: yy-mm-dd_N_XXX.mp3
## XXX= WR, LuA, 3JT-2

# mp4 to mp3
ls *.mp4 | sed 's/.mp4//' | xargs -i -n 1 -P 7 ffmpeg -i {}.mp4 -vn -acodec libmp3lame -ac 2 -qscale:a 4 -ar 48000 {}.mp3


# Keep Original files
   mcd all
   cp ../*.mp3 .
   ls > move.sh
   # Bearbeiten der namen in Intellij mit <shift><alt><insert>


# adapt audio gain
ls *.mp3 | xargs -n 1 -P 7 mp3gain -r -k
----
