[source, javascript]
----
let p = new Promise((resolve, reject) => {
  let a = 1 + 1;
  if (a == 2) {
  resolve('Success');
  } else {
  reject('Failed');
  }
});

p.then((message) => {
  console.log('This is the then messsage: ' + message);
})
.catch((message) => {
  console.log('This is the catch message: ' + message);
})
.finally(() => {
  console.log('This is the finally message')
})
----
